function bootstrap-webpack.bootstrap_styles_loader (content)

  • description and source-code
    bootstrap_styles_loader = function (content) {
  this.cacheable(true);
  var config = this.exec(content, this.resourcePath);
  var start =
      "@import          \"~bootstrap/less/variables.less\";\n"
    + "@icon-font-path: \"~bootstrap/fonts/\";\n"
    + "@import          \"./bootstrap.config.less\";\n";
  source = start + styles.filter(function (style) {
    return config.styles[style];
  }).map(function (style) {
    return "@import \"~bootstrap/less/" + style + ".less\";";
  }).join("\n");
  return source;
}

function bootstrap-webpack.bootstrap_scripts_loader.pitch (configPath)

  • description and source-code
    pitch = function (configPath) {
  this.cacheable(true);
  var config = require(configPath);
  return scripts.filter(function (script) {
    return config.scripts[script];
  }).map(function (script) {
    return "require(" + JSON.stringify("imports-loader?jQuery=jquery!bootstrap/js/" + script) + ");";
  }).join("\n");
}
